+ * Run a PowerShell command as part of the kill-switch path with a hard
+ * timeout and guaranteed child cleanup. Replace any existing direct
+ * `execSync('powershell ...')` calls in the runtime with this helper.
+ */
+export function runKillSwitchPS(script: string, opts: KillSwitchOptions = {}): Promise<number> {
+  const timeoutMs = opts.timeoutMs ?? 5000;
+  return new Promise((resolve) => {
+    const child = spawn('powershell.exe', ['-NoProfile', '-NonInteractive', '-Command', script], {
+      stdio: 'ignore',
+      windowsHide: true,
+    });
+    let settled = false;
+    const timer = setTimeout(() => {
+      if (settled) return;
+      settled = true;
+      try { child.kill('SIGKILL'); } catch { /* ignore */ }
+      opts.onTimeout?.();
+      resolve(-1);
+    }, timeoutMs);
+    child.on('exit', (code) => {
+      if (settled) return;
+      settled = true;
+      clearTimeout(timer);
+      resolve(code ?? 0);
+    });
+    child.on('error', () => {
+      if (settled) return;
+      settled = true;
+      clearTimeout(timer);
+      resolve(-2);
+    });
