```diff
--- /dev/null
+++ b/scripts/lint_no_hardcoded_paths.mjs
@@
+#!/usr/bin/env node
+/**
+ * Portability lint: fails if any tracked source file contains a hardcoded
+ * absolute Windows user path. Wire into CI as `node scripts/lint_no_hardcoded_paths.mjs`.
+ */
+import { readFileSync, readdirSync, statSync } from 'node:fs';
+import path from 'node:path';
+
+const ROOT = path.resolve(process.cwd());
+const SKIP = new Set(['node_modules', '.git', 'dist', 'build', 'self_reports', 'LICENSE']);
+const EXTS = new Set(['.ts', '.tsx', '.js', '.mjs', '.cjs', '.json', '.md']);
+const PATTERNS = [
+  /[A-Za-z]:\\Users\\[A-Za-z0-9_.-]+/, // C:\Users\<name>
+  /\/home\/[A-Za-z0-9_.-]+\//,         // /home/<name>/
+];
+
+const offenders = [];
+
+function walk(dir) {
+  for (const entry of readdirSync(dir)) {
+    if (SKIP.has(entry)) continue;
+    const full = path.join(dir, entry);
+    const st = statSync(full);
+    if (st.isDirectory()) walk(full);
+    else if (EXTS.has(path.extname(entry))) check(full);
+  }
+}
+
+function check(file) {
+  // Skip self.
+  if (file.endsWith('lint_no_hardcoded_paths.mjs')) return;
+  const txt = readFileSync(file, 'utf8');
+  for (const re of PATTERNS) {
+    const m = txt.match(re);
+    if (m) offenders.push({ file: path.relative(ROOT, file), match: m[0] });
+  }
+}
+
+walk(ROOT);
+
+if (offenders.length) {
+  console.error('Hardcoded absolute paths detected:');
+  for (const o of offenders) console.error(`  ${o.file}: ${o.match}`);
+  process.exit(1);
+}
+console.log('OK: no hardcoded absolute user paths.');
```
